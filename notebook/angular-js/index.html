$injector

Simple

Run the code
  let module1 = angular.module('module1', []);
  module1.factory('dataService', () => {
    return {
      fn: () => 1,
    };
  });
  let injector = angular.injector(['module1']);
  injector.get('dataService').fn();

Double injector, shared view

Run the code
  function renderHtml(injector, template) {
    const $compile = injector.get('$compile');
    const link = $compile(template);

    const $rootScope = injector.get('$rootScope');
    const $scope = $rootScope.$new();

    const fragment = link($scope);
    $scope.$apply();

    return fragment.html();
  }

  let viewModule = angular.module('view', []);
  viewModule.component('viewer', {
    controller: function (dataService) {
      this.data = dataService.fn();
    },
    template: '<span>{{ $ctrl.data }}</span>'
  });

  let module1 = angular.module('module1', []);
  module1.factory('dataService', () => {
    return {
      fn: () => 1,
    };
  });
  let injector1 = angular.injector(['module1', 'ng', 'view']);

  let module2 = angular.module('module2', []);
  module2.factory('dataService', () => {
    return {
      fn: () => 2,
    };
  });
  let injector2 = angular.injector(['module2', 'ng', 'view']);

  [
    renderHtml(injector1, '<viewer></viewer>'),
    renderHtml(injector2, '<viewer></viewer>')
  ];

Overwriting services with loadNewModules

However, this is available only after 1.6.7, and in addition, won't override a service which has been requested already. After a service is instantiated, it's placed in cache, to which there's no access, and there's no code to alter its fields.
Run the code
  let module = angular.module('module', []);
  module.factory('dataService', () => {
    return {
      fn: () => 'original-value',
    };
  });

  let injector = angular.injector(['module']);
  injector.loadNewModules([($provide) => {
    $provide.factory('dataService', () => {
      return {
        fn: () => 'overwritten-value',
      };
    });
  }]);

  injector.get('dataService').fn();

  let injector = angular.injector(['module', 'module2']);
